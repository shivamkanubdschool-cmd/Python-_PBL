df_noisy = df.copy()
# 1. Add Gaussian Noise to numeric columns
num_cols = df_noisy.select_dtypes(include=[np.number]).columns
for col in num_cols:
    noise = np.random.normal(0, 0.1 * df_noisy[col].std(), df_noisy.shape[0])
    df_noisy[col] = df_noisy[col] + noise

# 2. Introduce Missing Values (5%)
for col in df_noisy.columns:
    df_noisy.loc[df_noisy.sample(frac=0.05).index, col] = np.nan

# 3. Add Duplicate Rows (5%)
df_noisy = pd.concat([df_noisy, df_noisy.sample(frac=0.05)], ignore_index=True)

# 4. Add Outliers
for col in num_cols:
    df_noisy.loc[df_noisy.sample(frac=0.02).index, col] *= 3
